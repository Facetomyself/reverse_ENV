在编译Chromium浏览器的时候，可以选择在win下编译x64还是x86，x86在编译时候速度会更快一些，体积稍小。
但是编译成功之后，查看堆栈入口，可以发现有一段专门的fiber调度：

  *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *


    int APIENTRY wWinMain(HINSTANCE instance, HINSTANCE prev, wchar_t*, int) {
    #if defined(ARCH_CPU_32_BITS)  ......  if (!::IsThreadAFiber()) {    constexpr size_t kStackSize = 4 * 1024 * 1024;  // 4 MiB    LPVOID original_fiber =        ::ConvertThreadToFiberEx(nullptr, FIBER_FLAG_FLOAT_SWITCH);    if (original_fiber) {      FiberState fiber_state = {instance, original_fiber};      LPVOID big_stack_fiber = ::CreateFiberEx(          0, kStackSize, FIBER_FLAG_FLOAT_SWITCH, FiberBinder, &fiber_state);      if (big_stack_fiber) {        ::SwitchToFiber(big_stack_fiber);        ::DeleteFiber(big_stack_fiber);        ::ConvertFiberToThread();        return fiber_state.fiber_result;      }    ......  }
    #endif  // defined(ARCH_CPU_32_BITS)

这段fiber调度实际调用：

  *   *   *   *   *   *


    void WINAPI FiberBinder(void* params) {  auto* fiber_state = static_cast<FiberState*>(params);  fiber_state->fiber_result =      wWinMain(fiber_state->instance, nullptr, nullptr, 0);  ::SwitchToFiber(fiber_state->original_fiber);}

由此可以引出一个轻量级线程的概念。

Windows 中有两种线程的概念：

  * ** 线程（Thread）  ** ：内核管理，切换成本较高。

  * ** Fiber  ** ：用户态调度（用户自己管理切换），线程内可以创建多个 Fiber，在同一个线程上切换。


对比项  |  Thread（线程）  |  Fiber（纤程）
---|---|---
切换方式  |  内核态调度  |  用户态切换
上下文  |  独立线程栈  |  共享线程上下文
开销  |  大  |  小
控制权  |  操作系统  |  程序员手动切换


为什么32位专门用了Fiber线程做调度  ？

Fiber 允许手动控制栈和上下文切换，这里解决了  ** 32 位 Windows 主线程栈小  ** 的问题，  ** 给主线程扩展了栈  ** ，避免
32 位 Chromium 在复杂操作中崩溃。在 64 位 Windows 上就不用了，因为默认 8MB 栈已经够大。

举个例子，V8执行 JS 代码时，每次函数调用都消耗栈，  递归  很深时栈空间快速消耗。32 位栈本来就只有 ~1MB，复杂页面 + 执行 JS
脚本就容易用尽栈。

  * ` CreateFiberEx  ` 可以指定  ** 大栈  ** ，比如 4MB（32 位）。

###  在看到chromium源码中使用它后，我们可以学习起来，在日常编码中引入Fiber，比如用户态主动使用Fiber进行指纹开发，  `
CreateFiberEx  ` 指定 4MB 栈，模拟 JS 复杂 Canvas 绘制：

  *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *   *


    #include <windows.h>#include <iostream>#include <vector>#include <string>
    struct FiberState {    LPVOID original_fiber;    int result;};
    // 简单的 Canvas 的模拟函数void drawCanvas(int width, int height, const std::string& text, std::vector<int>& buffer) {    buffer.resize(width * height);    for (int y = 0; y < height; ++y) {        for (int x = 0; x < width; ++x) {            int color = (x + y) % 256;             if (x > 10 && x < 10 + (int)text.size() && y == 10) {                color = 255;             }            buffer[y * width + x] = color;        }    }}
    // Fiber 执行函数void WINAPI CanvasFiber(void* param) {    FiberState* state = static_cast<FiberState*>(param);
        std::vector<int> canvas;    drawCanvas(200, 100, "Hello RUYI Browser", canvas);
        std::cout << "Canvas generated! Size: " << canvas.size() << std::endl;
        // 切回主线程    ::SwitchToFiber(state->original_fiber);}
    int main() {    // 先转 fiber    if (!IsThreadAFiber()) {        LPVOID mainFiber = ConvertThreadToFiber(nullptr);        if (!mainFiber) {            std::cerr << "ConvertThreadToFiber failed: " << GetLastError() << std::endl;            return 1;        }
            FiberState state;        state.original_fiber = mainFiber;
            // 创建大栈 Fiber        constexpr size_t kStackSize = 4 * 1024 * 1024; // 4MB        LPVOID canvasFiber = CreateFiberEx(            0, kStackSize, FIBER_FLAG_FLOAT_SWITCH, CanvasFiber, &state        );
            if (!canvasFiber) {            std::cerr << "CreateFiberEx failed: " << GetLastError() << std::endl;            return 1;        }
            // 切换到 Fiber 执行绘制        SwitchToFiber(canvasFiber);
            // Fiber 执行完毕，清理        DeleteFiber(canvasFiber);
            std::cout << "Back to main thread!" << std::endl;        ConvertFiberToThread();    }
        return 0;}
