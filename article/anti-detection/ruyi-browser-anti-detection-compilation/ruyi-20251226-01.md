不过最近自制CDP自动化框架，也发现CDP确实强大，只是被检测的点也很多，它被检测的点是自动化框架和人真实操作不同的地方，需要注意的颇多。不过用它来做指纹浏览器的启动器，是非常合适的：

1.用CDP做指纹浏览器给那些畏惧chromium源码的人开了一条研究指纹修改的新路。

2.用CDP做自动化框架异常简单，因为只是官方文档中的一条一条命令，任何Python新手都可以在一天内学会制作。

3.尤其是当下需要做指纹浏览器的定制化，CDP框架通过专门的JS脚本注入，可以达到比较理想的过检测效果。

CDP的编写远没有大家想的复杂，只是因为目前还未普及罢了，等到一两年后，我估计大家人人都会编写自己的自动化框架了。


    import requests
    import websocket
    import json
    import time
    import random

    id_counter = 0


    def send_cdp(ws, method, params=None):
    global id_counter
        current_id = id_counter
        ws.send(json.dumps({
    'id': current_id,
    'method': method,
    'params': params if params is not None else {}
        }))
        id_counter += 1

    # 循环接收，直到找到对应 ID 的响应，同时打印事件（如 Network.requestWillBeSent）
    while True:
            response = json.loads(ws.recv())
    if response.get('id') == current_id:
    return response


    # ===== 新增：模拟 iPhone 指纹函数 =====
    def set_iphone_chrome119_fingerprint(ws):
    """
    深度模拟 iPhone 上的 Chrome 119 浏览器指纹 (HTTP 层)
    目标：iOS 17.1, Chrome 119.0.6045.169
        """
    # 1. 定义标准的 Chrome for iOS (CriOS) User-Agent
        # 注意：iOS 上的 Chrome 必须包含 CriOS 关键字
    iphone_ua = "Mozilla/5.0 (iPhone; CPU iPhone OS 17_1 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) CriOS/119.0.6045.169 Mobile/15E148 Safari/604.1"

    # 2. 配置 Client Hints (sec-ch-ua 系列)
        # 这些决定了 HTTP 请求头中 sec-ch-ua-* 的具体值
    ua_metadata = {
    "brands": [
                {"brand": "Google Chrome", "version": "119"},
                {"brand": "Chromium", "version": "119"},
                {"brand": "Not?A_Brand", "version": "24"}
            ],
    "fullVersionList": [
                {"brand": "Google Chrome", "version": "119.0.6045.169"},
                {"brand": "Chromium", "version": "119.0.6045.169"},
                {"brand": "Not?A_Brand", "version": "24.0.0.0"}
            ],
    "fullVersion": "119.0.6045.169",
    "platform": "iOS",
    "platformVersion": "17.1.0",
    "architecture": "arm", # iPhone 是 ARM 架构
    "model": "iPhone",
    "mobile": True,
    "bitness": "",
    "wow64": False
    }

    # 3. 强制覆盖 User-Agent 和 Client Hints
    send_cdp(ws, "Network.setUserAgentOverride", {
    "userAgent": iphone_ua,
    "acceptLanguage": "zh-CN,zh;q=0.9,en-US;q=0.8,en;q=0.7",
    "platform": "iPhone",
    "userAgentMetadata": ua_metadata
        })

    # 4. 补充额外的 HTTP 头部 (确保这些在所有请求中强制一致)
        # 这能修改常规方法改不到的特定 Header
    extra_headers = {
    "sec-ch-ua": '"Google Chrome";v="119", "Chromium";v="119", "Not?A_Brand";v="24"',
    "sec-ch-ua-mobile": "?1",
    "sec-ch-ua-platform": '"iOS"',
    "sec-ch-ua-platform-version": '"17.1.0"',
    "sec-ch-ua-model": '"iPhone"',
    "Upgrade-Insecure-Requests": "1"
    }
        send_cdp(ws, "Network.setExtraHTTPHeaders", {"headers": extra_headers})

    print(" 已成功模拟 Chrome 119 (iOS 17.1) 深度 HTTP 指纹")


    # ===== 基础操作函数 =====
    def ruyi_navigator(ws, url):
    return send_cdp(ws, 'Page.navigate', {'url': url})


    # ===== 主流程 =====
    # 1. 连接浏览器
    try:
        url = "http://127.0.0.1:9222/json"
    pages = requests.get(url).json()
        ws_url = next((p['webSocketDebuggerUrl'] for p in pages if p.get('type') == 'page'), None)
        ws = websocket.create_connection(ws_url)
    except Exception as e:
    print(f" 无法连接到浏览器，请确保开启了 --remote-debugging-port=9222。错误: {e}")
    exit()

    # 2. 启用网络和 DOM 代理
    send_cdp(ws, 'Network.enable')


    set_iphone_chrome119_fingerprint(ws)


    print("\n-> 正在前往 http://127.0.0.1:8000 ...")
    # 4. 访问目标 URL
    ruyi_navigator(ws, "http://127.0.0.1:8000")

    # 5. 持续接收一段时间的包，以便我们在控制台看到 Network.requestWillBeSent 的打印
    print(" 监控中，观察上方打印的请求头内容...")
    try:
    for _ in range(20):  # 监听一段时间
    json.loads(ws.recv())  # 这会触发 send_cdp 循环中的打印逻辑
    time.sleep(0.5)
    except KeyboardInterrupt:
    pass

    ws.close()
