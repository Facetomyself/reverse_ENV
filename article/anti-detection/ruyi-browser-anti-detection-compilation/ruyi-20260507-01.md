如意最近又肝了一个补环境工具，核心是把  浏览器的JS引擎+全部的DOM环境拿了下来  ，再用对应的浏览器网络库请求即可。

不再使用node的vm2之类的这种沙盒，采用真实原生的浏览器JS引擎和环境来跑代码生成参数，核心的指纹开接口进行替换。

    #!/usr/bin/env python3"""fangdi.com.cn 瑞数 6 反爬 — 纯 ruyijs 补环境单文件采集脚本
        python fangdi_collector.py    python fangdi_collector.py --api /service/index/getWriteDict.action -o out.json
        from fangdi_collector import FangdiCollector    data = FangdiCollector().fetch_homepage()"""from __future__ import annotationsimport argparse, json, logging, re, sys, timefrom pathlib import Pathfrom typing import Any
    sys.path.insert(0, r"C:\firefox\ruyijs")import ruyijsfrom curl_cffi import requests as curl_requests
    BASE_URL = "https://www.fangdi.com.cn"DEFAULT_UA = "Mozilla/5.0 (Windows NT 10.0; Win64; x64; rv:151.0) Gecko/20100101 Firefox/151.0"LOG = logging.getLogger("fangdi")
    # ─── Challenge HTML 解析 ─────────────────────────────────────────
    _RE = {    "ts":   re.compile(r"<script[^>]*>\s*(\$_ts\s*=\s*window\['\$_ts'\].*?)</script>", re.DOTALL),    "path": re.compile(r'<script\s+[^>]*src="(/[A-Za-z0-9_-]+/[A-Za-z0-9._-]+\.js)"[^>]*\br\s*=\s*[\'"]m[\'"]'),    "meta": re.compile(r"""<meta\s+[^>]*\bid=["']([^"']+)["'][^>]*\bcontent=["']([^"']+)["'][^>]*\br\s*=\s*['"]m['"]"""),    "metaR":re.compile(r"""<meta\s+[^>]*\bcontent=["']([^"']+)["'][^>]*\bid=["']([^"']+)["'][^>]*\br\s*=\s*['"]m['"]"""),    "trig": re.compile(r"""<script[^>]*\br\s*=\s*['"]m['"][^>]*>\s*(_\$[A-Za-z0-9_]+\(\))\s*;?\s*</script>"""),}
    def _parse(html: str):    m = _RE["ts"].search(html)    if not m: return None    ts = m.group(1)    m = _RE["path"].search(html)    if not m: return None    path = m.group(1)    m = _RE["meta"].search(html)    if m: mid, mc = m.group(1), m.group(2)    else:        m = _RE["metaR"].search(html)        if not m: return None        mid, mc = m.group(2), m.group(1)    m = _RE["trig"].search(html)    return ts, path, mid, mc, (m.group(1) if m else None)
    # ─── DOM 注入 + RS JS 执行 ───────────────────────────────────────
    _DOM_JS = r"""(function(){ var m=document.createElement('meta'); m.setAttribute('id',%(mi)s);m.setAttribute('content',%(mc)s);m.setAttribute('r','m'); document.head.appendChild(m); var s=document.createElement('script'); s.setAttribute('type','text/javascript');s.setAttribute('r','m');s.text=%(ts)s; document.head.appendChild(s); var e=document.createElement('script'); e.setAttribute('type','text/javascript');e.setAttribute('charset','utf-8'); e.setAttribute('src',%(rp)s);e.setAttribute('r','m'); document.head.appendChild(e); var b=document.createElement('script'); b.setAttribute('type','text/javascript');b.setAttribute('r','m'); document.body.appendChild(b);})();"""
    def _run_challenge(session, ua, browser, impersonate, timeout):    """GET / → parse → run RS JS → return (cookies, ctx_or_None, status)."""    resp = session.get(BASE_URL + "/", timeout=timeout, allow_redirects=False)    LOG.info("  response: status=%d size=%d", resp.status_code, len(resp.text))    p = _parse(resp.text)    if not p:        raise RuntimeError(f"parse failed (status={resp.status_code})")    ts, rp, mi, mc, trig = p
        rs_js = session.get(BASE_URL + rp, timeout=timeout).text    seeded = {n: session.cookies.get(n) for n in session.cookies if session.cookies.get(n)}
        ctx = ruyijs.Context.for_anti_bot(        target_browser=browser, location=BASE_URL + "/",        user_agent=ua, impersonate=impersonate, seeded_cookies=seeded,    )    dom = _DOM_JS % {"mi": json.dumps(mi), "mc": json.dumps(mc), "ts": json.dumps(ts), "rp": json.dumps(rp)}    ctx.eval(dom, f"{BASE_URL}/dom-setup.js")    ctx.eval(ts, f"{BASE_URL}/")    ctx.eval(rs_js, f"{BASE_URL}{rp}")    if trig:        try: ctx.eval(trig, f"{BASE_URL}/trigger")        except Exception: pass    ctx.drain()    return ctx.get_cookies(), ctx, resp.status_code
    # ─── FangdiCollector ─────────────────────────────────────────────
    class FangdiCollector:    """    最简::        data = FangdiCollector().fetch_homepage()
        with 语句::        with FangdiCollector() as c:            home = c.fetch_homepage()            other = c.call_api("/service/index/getWriteDict.action")    """    def __init__(self, target_browser="firefox-151-windows", user_agent=None,                 impersonate="firefox147", timeout=30.0, retry=3, log_level="INFO"):        self.browser = target_browser        self.ua = user_agent or DEFAULT_UA        self.imp = impersonate        self.timeout = timeout        self.retry = retry        self._sess = self._ctx = None        self._closed = False        logging.basicConfig(level=getattr(logging, log_level.upper(), logging.INFO),                            format="%(asctime)s [%(levelname)s] %(name)s: %(message)s", datefmt="%H:%M:%S")
        def __enter__(self): return self    def __exit__(self, *a): self.close()
        def close(self):        if self._closed: return        self._closed = True        for o in (self._ctx, self._sess):            if o:                try: o.close()                except: pass        self._ctx = self._sess = None
        def fetch_homepage(self) -> dict[str, Any]:        return self.call_api("/service/index/getIndexMessage.action")
        def call_api(self, path: str) -> Any:        last = None        for i in range(1, self.retry + 1):            try:                return self._once(path)            except Exception as e:                last = e                LOG.warning("attempt %d/%d: %s", i, self.retry, e)                if i < self.retry:                    time.sleep(2 ** (i - 1))                    self._reset()        raise last
        def _reset(self):        for o in (self._ctx, self._sess):            if o:                try: o.close()                except: pass        self._ctx = self._sess = None
        def _once(self, api_path: str) -> Any:        if not self._sess:            self._sess = curl_requests.Session(impersonate=self.imp)            self._sess.headers.update({"User-Agent": self.ua})
            # Phase 1        LOG.info("Phase 1: challenge")        c1, ctx1, _ = _run_challenge(self._sess, self.ua, self.browser, self.imp, self.timeout)        ctx1.close()        for n, v in c1.items():            self._sess.cookies.set(n, v, domain="www.fangdi.com.cn")
            # Phase 2        LOG.info("Phase 2: real page")        c2, ctx2, st2 = _run_challenge(self._sess, self.ua, self.browser, self.imp, self.timeout)        if st2 != 200:            ctx2.close()            raise RuntimeError(f"Phase 2: expected 200, got {st2}")        self._ctx = ctx2
            # Phase 3        LOG.info("Phase 3: XHR %s", api_path)        self._ctx.drain(timeout_ms=1000)        r = self._ctx.xhr("POST", api_path, headers={            "Content-Type": "application/x-www-form-urlencoded;charset=utf-8",            "Accept": "application/json, text/javascript, */*; q=0.01",            "X-Requested-With": "XMLHttpRequest",        })        LOG.info("  XHR: status=%s url=%s", r.get("status"), r.get("url", "")[:120])        if not r.get("ok") or r.get("status") != 200:            raise RuntimeError(f"Phase 3 failed: status={r.get('status')} {r.get('error','')}")        return json.loads(r["text"])
    # ─── CLI ─────────────────────────────────────────────────────────
    def main() -> int:    ap = argparse.ArgumentParser(description="fangdi.com.cn 瑞数6 ruyijs采集",                                 formatter_class=argparse.ArgumentDefaultsHelpFormatter)    ap.add_argument("--api", default="/service/index/getIndexMessage.action")    ap.add_argument("-o", "--output", default="homepage_data.json")    ap.add_argument("--target-browser", default="firefox-151-windows")    ap.add_argument("--timeout", type=float, default=30.0)    ap.add_argument("--retry", type=int, default=3)    ap.add_argument("--log-level", default="INFO", choices=["DEBUG","INFO","WARNING","ERROR"])    a = ap.parse_args()
        try:        with FangdiCollector(a.target_browser, timeout=a.timeout, retry=a.retry, log_level=a.log_level) as c:            data = c.call_api(a.api)    except Exception as e:        LOG.error("FAIL: %s", e, exc_info=True)        return 1
        Path(a.output).write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")    LOG.info("Saved to %s", a.output)    if isinstance(data, dict):        LOG.info("签约: %s㎡ / %s套, %d个区",                 data.get("todaySign_area","?"), data.get("todaySign_ts","?"),                 len(data.get("housingSupply",[])))    return 0
    if __name__ == "__main__":    sys.exit(main())
