# 逆向某音乐 App：用 Frida + unidbg 还原 AES+RSA 混合加密的设备指纹 dfid

> 来源: 微信公众号：泡泡以安

>  dfid 是某音乐 App 设备注册协议（ rregisterdev ）服务端返回的设备唯一标识符。本文通过静态分析与动态插桩相结合的方式，完整还原了该注册协议中涉及的 AES + RSA 混合加密 方案，包括。

##  概述

` dfid  ` 是某音乐 App 设备注册协议（  ` r_register_dev  `
）服务端返回的设备唯一标识符。本文通过静态分析与动态插桩相结合的方式，完整还原了该注册协议中涉及的  ** AES + RSA 混合加密  **
方案，包括：

  * 请求体（  ` POST Body  ` ）的 AES-CBC 加密逻辑
  * URL 参数  ` p  ` 的 RSA 公钥加密逻辑
  * AES 密钥（key + iv）的动态派生机制

** 不涉及范围  ** ：  ` mid  ` 、  ` signature  ` 等参数（可在 Java 层直接读取，感兴趣的读者可自行分析）。

###  1.2 加密参数识别

经过分析，该请求中存在以下加密/签名字段：

参数位置  |  字段名  |  特征
---|---|---

本文重点分析：  ** 参数  ` p  ` ** 与  ** 请求体  ** 的加密还原。

###  2.4 通过 Hook RegisterNatives 定位 SO 文件

利用 Frida hook  ` libart.so  ` 中的  ` RegisterNatives  ` 函数，可以追踪到 native
方法的注册信息：


    // 参考：https://github.com/lasting-yang/frida_hook_libart
    function find_RegisterNatives(params) {
    let symbols = Module.enumerateSymbolsSync("libart.so");
    let addrRegisterNatives = null;
    for (let i = 0; i < symbols.length; i++) {
        let symbol = symbols[i];
        if (symbol.name.indexOf("art") >= 0 &&
            symbol.name.indexOf("JNI") >= 0 &&
            symbol.name.indexOf("RegisterNatives") >= 0 &&
            symbol.name.indexOf("CheckJNI") < 0) {
          addrRegisterNatives = symbol.address;
          console.log("RegisterNatives is at ", symbol.address, symbol.name);
          hook_RegisterNatives(addrRegisterNatives);
        }
      }
    }

    function hook_RegisterNatives(addrRegisterNatives) {
    if (addrRegisterNatives != null) {
        Interceptor.attach(addrRegisterNatives, {
          onEnter: function (args) {
            console.log("[RegisterNatives] method_count:", args[3]);
            let java_class = args[1];
            let class_name = Java.vm.tryGetEnv().getClassName(java_class);
            let methods_ptr = ptr(args[2]);
            let method_count = parseInt(args[3]);

            for (let i = 0; i < method_count; i++) {
              // 每个 JNI 方法注册项由 3 个指针组成：name / sig / fnPtr
              let name_ptr = Memory.readPointer(methods_ptr.add(i * 3 * Process.pointerSize));
              let sig_ptr = Memory.readPointer(methods_ptr.add((i * 3 + 1) * Process.pointerSize));
              let fnPtr_ptr = Memory.readPointer(methods_ptr.add((i * 3 + 2) * Process.pointerSize));

              let name = Memory.readCString(name_ptr);
              let sig = Memory.readCString(sig_ptr);
              letmodule = Process.findModuleByAddress(fnPtr_ptr);

              console.log("[RegisterNatives] java_class:", class_name, "name:", name,
                "sig:", sig, "fnPtr:", fnPtr_ptr,
                "module_name:", module.name, "module_base:", module.base,
                "offset:", ptr(fnPtr_ptr).sub(module.base));
            }
          }
        });
      }
    }

    setImmediate(find_RegisterNatives);

编写 Frida 脚本同时 hook  ` AES_cbc_encrypt  ` 和  ` AES_set_encrypt_key  ` 函数：


    var arg1, arg2, arg3, arg4, arg5;

    // Hook AES_cbc_encrypt（偏移 0x3583C）
    function inline_hook_3583C() {
        var libnative_lib_addr = Module.findBaseAddress("libcrypto_kg.so");
        if (libnative_lib_addr) {
            // Thumb 模式下地址需 +1
            var addr_3583C = libnative_lib_addr.add(0x3583D);
            console.log("addr_3583C:", addr_3583C);

            var ptr_func = new NativePointer(addr_3583C);
            Interceptor.attach(ptr_func, {
                onEnter: function (args) {
                    arg1 = args[0]; // 明文输入
                    arg2 = args[1]; // 密文输出缓冲区
                    arg3 = args[2]; // 数据长度

###  2.11 关键发现：Key 和 IV 是变化的

通过多次抓包与 hook 验证，发现 AES 的 Key 和 IV  ** 每次请求都不同  ** 。这意味着它们不是硬编码的，而是动态派生的。

** 阶段性小结  ** ：

> 请求体 = AES-CBC 加密（设备指纹 JSON）
>  AES Key 和 IV 动态变化，与参数  ` p  ` 相关
>  参数  ` p  ` 长度固定为 256 hex 字符，典型 RSA-1024 加密结果

这是经典的  ** AES + RSA 混合加密  ** 方案：RSA 加密随机密钥，AES 加密实际数据。

        def __init__(self):

###  AES 参数派生


    import hashlib

    def derive_aes_params(random_str: str) -> tuple:
        """
        从随机字符串派生 AES Key 和 IV
        :param random_str: RSA 明文中的 aes 字段值，如 "fE6l15"
        :return: (key_hex_16chars, iv_hex_16chars)
        """
        # MD5 哈希后取 32 个 hex 字符
        md5_hex = hashlib.md5(random_str.encode()).hexdigest()
        aes_key = md5_hex[:16]   # 前 16 字符作为 AES Key
        aes_iv  = md5_hex[16:]   # 后 16 字符作为 AES IV
        return aes_key, aes_iv

    # 示例验证
    key, iv = derive_aes_params("cr3qJ8")
